\subsection{Creating Vector Images with TikZ}
\begin{frame}{Enhancing Visual Quality: Fast Vector Graphics with AI}
    \begin{itemize}
        \setlength{\itemsep}{14pt}
        \item \textbf{Fast \& Professional Figures}: AI allows you to generate high-quality illustrations quickly without wasting time.
        \item \textbf{What is TikZ?}: A native LaTeX programming language for vector graphics that integrates perfectly with your document.
        \item \textbf{Full Control}: Since TikZ figures are code-based, you always keep full control to manually edit or fine-tune any detail at any time.
    \end{itemize}
\end{frame}

            \textbf{TikZ Code Structure}:
            \begin{block}{}
                \texttt{\scriptsize
                \textbackslash begin\{tikzpicture\}[scale=0.45] \\
                \ \ \textbackslash tikzset\{declare function=\{f(x) = ...\}\} \\
                \ \ \\
                \ \ \% Draw density curve \\
                \ \ \textbackslash draw[thick, blue] plot[domain=0:8] ...; \\
                \ \ \\
                \ \ \% Draw cluster tree bars \\
                \ \ \textbackslash draw[very thick, red] (1.87,1.70) -- ...; \\
                \ \ \\
                \ \ \% Add labels \\
                \ \ \textbackslash draw[<-, semithick] (1.87,1.82) -- ...; \\
                \textbackslash end\{tikzpicture\}
                }
            \end{block}
        \end{column}
        \begin{column}{0.42\textwidth}
            \textbf{LaTeX Integration}:
            \begin{itemize}
                \setlength{\itemsep}{12pt}
                \item \textbf{Include Package}:
                \begin{itemize}
                    \item Add \texttt{\textbackslash usepackage\{tikz\}} in your preamble.
                \end{itemize}
                \item \textbf{Direct Placement}:
                \begin{itemize}
                    \item Paste the generated code directly in the document body.
                \end{itemize}
                \item \textbf{No Assets to Manage}:
                \begin{itemize}
                    \item No need for external image files (PNG/PDF).
                \end{itemize}

\begin{frame}{Case Study: The Long-Path Intuition}
    \begin{center}
        \begin{tikzpicture}[scale=1.2, every node/.style={circle, draw, minimum size=0.45cm, inner sep=0.5pt, font=\scriptsize\bfseries}]
            \node[fill=inria-bleu!30, draw=inria-bleu] (x) at (0,0) {$x (+)$};
            \node[fill=inria-bleu!20, draw=inria-bleu] (v1) at (1.5,0) {$v_1 (+)$};
            \node[fill=inria-bleu!20, draw=inria-bleu] (v2) at (3.0,0) {$v_2 (+)$};
            \node[fill=inria-bleu!20, draw=inria-bleu] (v3) at (4.5,0) {$v_3 (+)$};
            \node[fill=inria-bleu!30, draw=inria-bleu] (y) at (6.0,0) {$y (+)$};

            \draw[very thick, black] (x) -- (v1) node[midway, above, font=\tiny] {$+$};
            \draw[thin, dashed, black] (v1) -- (v2) node[midway, above, font=\tiny] {$-$};
            \draw[very thick, black] (v2) -- (v3) node[midway, above, font=\tiny] {$+$};
            \draw[thin, dashed, black] (v3) -- (y) node[midway, above, font=\tiny] {$-$};
        \end{tikzpicture}
    \end{center}
    \textbf{The Path Parity Argument}:
    \begin{itemize}
        \setlength{\itemsep}{6pt}
        \item Consider two distant nodes $x, y$ both in community $+$.
        \item For any path of length $n$, each edge is corrupted (antiferromagnetic) with probability $1-p$.
        \item Parity is conserved: $x$ and $y$ are inferred in the same community only if the path has an \textbf{even} number of antiferromagnetic edges.
        \item As $n \to \infty$, the probability of an odd number of antiferromagnetic edges:
        \[ \mathbb{P}(\text{odd}) \longrightarrow \frac{1}{2} \]
        \item This washes out the community correlation, leading to the false intuition that weak recovery is impossible for any $p < 1$.
    \end{itemize}
\end{frame}

\begin{frame}{Case Study: Why MAP Works (Peierls Contours)}
    \textbf{Proof Sketch by Peierls Argument}:
    \begin{itemize}
        \setlength{\itemsep}{6pt}
        \item \textbf{Contours}: Boundaries of incorrect spin decisions form contours $\gamma$ on the dual grid.
        \item \textbf{MAP Optimality}: Any contour $\gamma$ in the MAP estimator must contain at least 50\% corrupted edges (otherwise, flipping the region inside would improve the MAP score):
        \[ B(\gamma) \ge \frac{|\gamma|}{2} \]
        \item \textbf{Chernoff Bound}: The probability of a fixed contour $\gamma$ of length $\ell$ being incorrect is bounded by $(2\sqrt{p(1-p)})^\ell$.
        \item \textbf{Contour Counting}: The number of contours of length $\ell$ around any node is at most $\ell 3^\ell$. The probability of an incorrect node is bounded by:
        \[ \sum_{\ell \ge 4} \ell (6\sqrt{p(1-p)})^\ell \xrightarrow[p \to 1]{} 0 \]
        \item \textbf{Conclusion}: Weak recovery is achieved for $p > p_0$, saving months of trying to prove a false statement.
    \end{itemize}
\end{frame}
