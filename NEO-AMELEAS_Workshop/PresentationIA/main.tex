\begin{frame}{Clustering on Euclidean Data $\mathcal{X} \subset \mathbb{R}^d$: Density Estimator}
    \begin{center}
        \Large $\widehat{f} \leftarrow \widehat{f}_{K-\text{NN}}$ \\[4pt]
        \normalsize \textbf{Estimator}: $K$-Nearest Neighbors ($K$-NN)
    \end{center}
    
    \vspace{0.4cm}
    
    \begin{columns}[c, onlytextwidth]
        \begin{column}{0.56\textwidth}
            \begin{center}
                \vspace{0.2cm}
                {\Large\mathversion{bold} $\forall x \in \mathbb{R}^d, \quad \widehat{f}_{K-\text{NN}}(x) = \frac{K}{n \, \omega_d \, r_K(x)^d}$}
                
                \vspace{1.0cm}
                
                {\large\mathversion{bold} $r_K(x) = \min \left\{ r \ge 0 \;\middle|\; |B(x,r) \cap \mathcal{X}| \ge K \right\}$}
            \end{center}
        \end{column}
        
        \begin{column}{0.42\textwidth}
            \begin{itemize}
                \setlength{\itemsep}{8pt}
                \item \textbf{$n$}: number of points in $\mathcal{X}$
                \item \textbf{$\omega_d$}: volume of the unit ball in $\mathbb{R}^d$
                \item \textbf{$r_K(x)$}: distance to the $K$-th nearest neighbor of $x$
                \item \textbf{$\omega_d r_K(x)^d$}: volume of the ball containing the $K$ neighbors
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

        \begin{column}{0.48\textwidth}
            \textbf{Chaining effect}:
            Classic Single-Linkage ($1$-NN) is highly sensitive to noise that artificially connects two clusters.
            
            \vspace{0.4cm}
            \textbf{Robust Single-Linkage} \cite{RobustSingleLinkage2010}:
            \begin{itemize}
                \setlength{\itemsep}{6pt}
                \item Uses the \textbf{core distance} $r_K(x)$ (radius of the $K$-NN) as a local density estimator.
                \item Replaces the geometric distance with the \textbf{mutual reachability distance} $d_{\mathrm{mreach}, K}$.
                \item Pushes isolated noise points away by artificially increasing their distance to other points.
            \end{itemize}
        \end{column}

\begin{frame}{From Robust Single-Linkage to (H)DBSCAN}
    \textbf{Limitation of Robust Single-Linkage}:
    \begin{itemize}
        \setlength{\itemsep}{6pt}
        \item Too restrictive for points located at the periphery of the clusters.
        \item Excludes border points that do not have $K$ neighbors, which reduces cluster recall by unduly discarding them as noise.
    \end{itemize}
    
    \vspace{0.4cm}
    \textbf{DBSCAN} \cite{DBSCAN} (Ester \emph{et al.}, 1996):
    \begin{itemize}
        \setlength{\itemsep}{6pt}
        \item Introduces a distinction between \textbf{core points} ($r_K(x) \le r$) and \textbf{border points} (reachable from a core, but without having $K$ neighbors themselves).
        \item Allows much better classification of cluster boundaries, hence the name (\emph{Applications with Noise}).
    \end{itemize}
    
    \vspace{0.4cm}
    \begin{block}{From DBSCAN to HDBSCAN}
        DBSCAN requires a single global threshold $r$. 
        \textbf{HDBSCAN} \cite{HDBSCAN} (Campello \emph{et al.}, 2013) continuously varies this threshold to eliminate this parameter and build a complete hierarchy.
    \end{block}
\end{frame}

\begin{frame}{The Case $K=1$: The \textsc{Gabriel} Graph}
    \begin{columns}[c, onlytextwidth]

                \item \textbf{Definition}:
                \begin{itemize}
                    \item An edge $[A, B]$ is a \textsc{Gabriel} edge if its \textbf{diametral disk} does not contain any other point in $\mathcal{X}$.

\begin{frame}{Generalization to Order $K$: Separating Simplices}
    \begin{columns}[c, onlytextwidth]
        \begin{column}{0.54\textwidth}
            \begin{itemize}
                \setlength{\itemsep}{10pt}
                \item \textbf{$K$-separating simplex}:
                \begin{itemize}
                    \item It is a $K$-simplex that has a \textbf{real influence} (causes clusters to merge) on the $K$-NN hierarchy.
                    \item \textbf{Case $K=1$}: $1$-separating simplices are precisely the \textbf{edges of the MST}.
                \end{itemize}
                \item \textbf{\textsc{Gabriel} condition}:
                \begin{itemize}
                    \item \textbf{Theorem}: any $K$-separating simplex is necessarily a \textbf{\textsc{Gabriel} simplex} ($\mathring{B}_\sigma \cap (\mathcal{X} \setminus \sigma) = \varnothing$).
                    \item \textbf{Computational domain}: the \textbf{higher-order \textsc{Delaunay} mosaic} \cite{osangEdelsbrunner2023algorithm}.
                \end{itemize}

\begin{frame}{The $K$-Minimum Spanning Tree Analogy}
    \begin{columns}[c, onlytextwidth]
        \begin{column}{0.54\textwidth}
            Similar to $K=1$, the higher-order hierarchy relies on a sparse combinatorial structure:
            
            \vspace{0.3cm}
            \begin{itemize}
                \setlength{\itemsep}{10pt}
                \item \textbf{Percolation objects}:
                \begin{itemize}
                    \item It is no longer points, but \textbf{facets} ($(K-1)$-simplices) that merge.
                \end{itemize}
                \item \textbf{\textsc{Gabriel} simplices}:
                \begin{itemize}
                    \item The simplices causing the merges (separating simplices) are \textbf{\textsc{Gabriel}} simplices.
                \end{itemize}
                \item \textbf{Computational domain}:
                \begin{itemize}
                    \item The search is restricted to the simplices of the \textbf{higher-order \textsc{Delaunay} mosaic} \cite{osangEdelsbrunner2023algorithm}.
                \end{itemize}
            \end{itemize}
        \end{column}
        
        \begin{column}{0.44\textwidth}
            \begin{center}
                \textbf{Comparison $1$-MST vs. $K$-MST}
                
                \vspace{0.5cm}
                \small
                \renewcommand{\arraystretch}{1.5}
                \begin{tabular}{lcc}
                    \toprule
                    & \textbf{$K = 1$} & \textbf{$K \ge 2$} \\
                    \midrule
                    \textbf{Vertices} & Points & $(K-1)$-simplices \\
                    \textbf{Merges} & Edges & $K$-simplices \\
                    \textbf{Condition} & \textsc{Gabriel} & \textsc{Gabriel} \\

\end{frame}

\begin{frame}{Summary: A Missing Link for Geometry Processing}
    \begin{itemize}
        \setlength{\itemsep}{12pt}
        \item \textbf{Higher-order \textsc{Delaunay} mosaic}:
        \begin{itemize}
            \item Contains the \textbf{entirety} of the hierarchical $K$-NN information.
            \item Represents the optimal and sparse support for higher-order clustering.
        \end{itemize}
        
        \item \textbf{Absence in existing libraries}:
        \begin{itemize}
            \item \textbf{Observation}: To date, there is \textbf{no geometry processing library} (such as CGAL or Geogram) integrating this object.
            \item \textbf{Consequence}: Need to develop complex \emph{ad-hoc} code to construct and manipulate these cell complexes.
        \end{itemize}
    \end{itemize}
\end{frame}
