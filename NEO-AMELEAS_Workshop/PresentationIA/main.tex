\subsection{Creating Vector Images with TikZ}
\begin{frame}{Enhancing Visual Quality: Fast Vector Graphics with AI}
    \begin{itemize}
        \setlength{\itemsep}{14pt}
        \item \textbf{Fast \& Professional Figures}: AI allows you to generate high-quality illustrations quickly without wasting time.
        \item \textbf{What is TikZ?}: A native LaTeX programming language for vector graphics that integrates perfectly with your document.
        \item \textbf{Full Control}: Since TikZ figures are code-based, you always keep full control to manually edit or fine-tune any detail at any time.
    \end{itemize}
\end{frame}

\begin{frame}{Example: Target Raster Image}
    \begin{columns}[c, onlytextwidth]
        \begin{column}{0.62\textwidth}
            \begin{center}
                \includegraphics[width=\textwidth]{imgs/EoM_HDBSCAN.png}
            \end{center}
        \end{column}
        \begin{column}{0.35\textwidth}
            \textbf{The Challenge}:
            \begin{itemize}
                \setlength{\itemsep}{12pt}
                \item A 1D density plot illustrating the \emph{excess of mass} in HDBSCAN.
                \item Simple raster formats (PNG/JPG) are pixelated and cannot match document fonts.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Example: Prompting the AI}
    \begin{columns}[c, onlytextwidth]
        \begin{column}{0.62\textwidth}
            \begin{center}
                \includegraphics[width=\textwidth]{imgs/Prompt_EoM_HDBSCAN.png}
            \end{center}
        \end{column}
        \begin{column}{0.35\textwidth}
            \textbf{Human-AI Collaboration}:
            \begin{itemize}
                \setlength{\itemsep}{12pt}
                \item Simply upload the target image and ask the AI to reproduce it in TikZ.
                \item Saves hours of calculating coordinates and writing syntax manually.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Example: The Generated TikZ Code}
    \begin{columns}[c, onlytextwidth]
        \begin{column}{0.55\textwidth}
            \textbf{TikZ Code Structure}:
            \begin{block}{}
                \texttt{\tiny
                \textbackslash begin\{tikzpicture\}[scale=0.35] \\
                \ \ \textbackslash tikzset\{declare function=\{f(x) = ...\}\} \\
                \ \ \\
                \ \ \% Draw density curve \\
                \ \ \textbackslash draw[thick, blue] plot[domain=0:8] ...; \\
                \ \ \\
                \ \ \% Draw cluster tree bars \\
                \ \ \textbackslash draw[very thick, red] (1.87,1.70) -- ...; \\
                \ \ \\
                \ \ \% Add labels \\
                \ \ \textbackslash draw[<-, semithick] (1.87,1.82) -- ...; \\
                \textbackslash end\{tikzpicture\}
                }
            \end{block}
        \end{column}
        \begin{column}{0.42\textwidth}
            \textbf{LaTeX Integration}:
            \begin{itemize}
                \setlength{\itemsep}{12pt}
                \item \textbf{Include Package}:
                \begin{itemize}
                    \item Add \texttt{\textbackslash usepackage\{tikz\}} in your preamble.
                \end{itemize}
                \item \textbf{Direct Placement}:
                \begin{itemize}
                    \item Paste the generated code directly in the document body.
                \end{itemize}
                \item \textbf{No Assets to Manage}:
                \begin{itemize}
                    \item No need for external image files (PNG/PDF).
                \end{itemize}
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}
