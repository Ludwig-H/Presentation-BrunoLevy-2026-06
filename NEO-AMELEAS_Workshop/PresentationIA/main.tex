\subsection{Creating Vector Images with TikZ}
\begin{frame}{Enhancing Visual Quality: Fast Vector Graphics with AI}
    \begin{itemize}
        \setlength{\itemsep}{14pt}
        \item \textbf{Fast \& Professional Figures}: AI allows you to generate high-quality illustrations quickly without wasting time.
        \item \textbf{What is TikZ?}: A native LaTeX programming language for vector graphics that integrates perfectly with your document.
        \item \textbf{Full Control}: Since TikZ figures are code-based, you always keep full control to manually edit or fine-tune any detail at any time.
    \end{itemize}
\end{frame}

\begin{frame}{Case Study: The Long-Path Intuition}
    \begin{center}
        \begin{tikzpicture}[scale=0.58, every node/.style={circle, draw, minimum size=0.4cm, inner sep=0.5pt, font=\scriptsize\bfseries}]
            \node[fill=inria-bleu!30, draw=inria-bleu] (x) at (0,0) {$+$};
            \node[fill=inria-bleu!30, draw=inria-bleu] (v1) at (2,0) {};
            \node[fill=inria-bleu!30, draw=inria-bleu] (v2) at (4,0) {};
            
            % Suspension dots as a node without border
            \node[draw=none, minimum size=0cm, inner sep=0pt] (dots) at (5.5,0) {\large $\dots$};
            
            \node[fill=inria-bleu!30, draw=inria-bleu] (vn1) at (7,0) {};
            \node[fill=inria-bleu!30, draw=inria-bleu] (y) at (9,0) {$+$};

            \draw[very thick, black] (x) -- (v1);
            \draw[thin, dashed, black] (v1) -- (v2);
            \draw[thin, black] (v2) -- (4.8,0);
            \draw[thin, black] (6.2,0) -- (vn1);
            \draw[thin, dashed, black] (vn1) -- (y);
        \end{tikzpicture}
    \end{center}
    \textbf{The Correlation Decay Dilemma}:
    \begin{itemize}
        \setlength{\itemsep}{6pt}
        \item For two distant $+$-nodes $x, y$ at distance $n$, we infer them as same-community ($+ +$) only if the path has an \textbf{even} number of antiferromagnetic edges.
        \item As $n \to \infty$, the probability that the path's sign product matches this true label ($+ +$) washes out:
        \[ \mathbb{P}(\text{even}) \xrightarrow[n \to \infty]{} \frac{1}{2} \]
        \item \textbf{The False Conjecture}: Simulations on increasingly larger grids showed the overlap vanishing. This intuition led me to question if weak recovery was truly possible for any noise level $p_c < 1$.
        \item \emph{The Catch}: The MCMC cluster dynamics simply had an extremely long mixing-time on larger grids, masking the true static phase.
    \end{itemize}
\end{frame}

\begin{frame}{Case Study: Why MAP Works (Peierls Contours)}
    \begin{columns}[c, onlytextwidth]
        \begin{column}{0.42\textwidth}
            \begin{center}
                \begin{tikzpicture}[scale=1.0, every node/.style={circle, draw, minimum size=0.35cm, inner sep=0pt, font=\tiny\bfseries}]
                    % Grid of nodes
                    \foreach \x in {0,1,2,3} {
                        \foreach \y in {0,1,2} {
                            % Determine if inside error contour (1,1) or (2,1)
                            \pgfmathsetmacro{\iserr}{0}
                            \ifnum\x=1 \ifnum\y=1 \gdef\iserr{1}\fi\fi
                            \ifnum\x=2 \ifnum\y=1 \gdef\iserr{1}\fi\fi
                            
                            \ifnum\iserr=1
                                \node[fill=inria-rouge!30, draw=inria-rouge] (N-\x-\y) at (\x,\y) {$-$};
                            \else
                                \node[fill=inria-bleu!30, draw=inria-bleu] (N-\x-\y) at (\x,\y) {$+$};
                            \fi
                        }
                    }
                    
                    % Draw grid edges
                    \foreach \x in {0,1,2,3} {
                        \foreach \y in {0,1,2} {
                            \ifnum\x<3
                                \draw[gray!40, thin] (\x,\y) -- (\x+1,\y);
                            \fi
                            \ifnum\y<2
                                \draw[gray!40, thin] (\x,\y) -- (\x,\y+1);
                            \fi
                        }
                    }
                    
                    % Draw the dual contour in orange/red
                    \draw[orange, line width=1.5pt, dashed] (0.5,0.5) -- (2.5,0.5) -- (2.5,1.5) -- (0.5,1.5) -- cycle;
                    \node[draw=none, orange, font=\scriptsize\bfseries] at (1.5, 1.7) {Contour $\gamma$};
                    
                    % Highlight the crossed edges (boundary edges) as bold
                    \draw[very thick, black] (1,0) -- (1,1);
                    \draw[very thick, black] (2,0) -- (2,1);
                    \draw[very thick, black] (0,1) -- (1,1);
                    \draw[very thick, black] (3,1) -- (2,1);
                    \draw[very thick, black] (1,2) -- (1,1);
                    \draw[very thick, black] (2,2) -- (2,1);
                \end{tikzpicture}
            \end{center}
        \end{column}
        \begin{column}{0.54\textwidth}
            \textbf{Peierls Contour Argument}:
            \begin{itemize}
                \setlength{\itemsep}{6pt}
                \item \textbf{Contours}: Boundaries separating correct (blue $+$) and incorrect (red $-$) spin decisions form contours $\gamma$ on the dual grid.
                \item \textbf{MAP Optimality}: The Maximum A Posteriori (MAP) estimator must have a majority of noisy edges along any contour $\gamma$:
                \[ B(\gamma) \ge \frac{|\gamma|}{2} \]
                \item Otherwise, flipping all spins inside the contour would improve the posterior MAP score.
                \item \textbf{Resolution}: The probability of having a majority-corrupted contour of length $\ell$ converges to 0 as the noise decreases ($p \to 1$).
                \item This proves that weak recovery is indeed achieved for $p > p_0$.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}
